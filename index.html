                    Step3:- Change To Your Local Time Zone tab To Finish

                    Step4:- Reboot The OS

                    Step5:- 
                    Hardware Setup:- 
                            1) Connect 4-digit 7 segment display to Raspberry Pi GPIO pins.
                                a) Connect The CLK To GPIO21
                                b) Connect The DIO pin To GPIO20
                                c) Connect The Vcc pin to 3.3v
                                d) Connect The GND pin to Ground Pin

                    Step7:- Run The Code and Observe the time which will be in HH:MM Format.

        Code:- 
from time import sleep
import tml637
        
    try: 
        import thread
    except ImportError:
        import _thread as thread
        
        # Initialize the TML637 display
    Display = tml637.TML637(CLK=21, DIO=20, Brightness=1)
        
    try:
        print("Starting clock in background (press CTRL+C to stop):")
            
        Display.startClock(military_time=False)
        
        print("Continue Python script while the clock is displayed")
            sleep(5)
        
            
        Display.ShowDoublePoint(False)
        sleep(5)
        
           
        loops = 3
        while loops > 0:
            for i in range(0, 10):
                Display.setBrightness(i | 100)
                sleep(0.5)
            loops -= 1
        
            
        Display.stopClock()
            
        thread.interrupt_main()
        
    except KeyboardInterrupt:
        print("Properly closing the clock and cleaning up GPIO pins.")
            
        Display.cleanup()

import time
import telepot
import RPi.GPIO as GPIO
    
    # GPIO setup
GPIO.setmode(GPIO.BOARD)
GPIO.setup(11, GPIO.OUT)
    
def handle(msg):
    chat_id = msg['chat']['id']
    command = msg['text'].lower()
    
    if command == "on":
        GPIO.output(11, GPIO.HIGH)
        bot.sendMessage(chat_id, "LED is ON")
    elif command == "off":
        GPIO.output(11, GPIO.LOW)
        bot.sendMessage(chat_id, "LED is OFF")
    else:
        bot.sendMessage(chat_id, "Use 'on' or 'off'")
    
    # Initialize bot
bot = telepot.Bot("Your Bot Token Here")
bot.message_loop(handle)
print("Listening...")
